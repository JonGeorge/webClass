How to Bundle JS and CSS using JAWR with the Spring Integration
---------------------------------------------------------------


Assumptions
-----------
 A) You have a Spring MVC Web App
 


Procedure
---------
 1. Add this dependency to your pom.xml
 
       <dependency>
          <groupId>net.jawr.extension</groupId>
          <artifactId>jawr-spring-extension</artifactId>
          <version>3.8</version>
       </dependency>
       

 2. Define a jawrControllerForJs in your application-context.xml
        <bean abstract="true" id="jawrBaseBean"  class="net.jawr.web.servlet.JawrSpringController">
           <property name="mapping" value=""/>
           <property name="controllerMapping" value="/jawr/"/>
           <property name="configLocation" value="/jawr.properties"/>
           <property name="configuration">
              <props>
                 <prop key="jawr.debug.on">true</prop>
              </props>
           </property>
        </bean>
        
        <bean id="jawrControllerForJs" parent="jawrBaseBean" />
        
        
 3. Update your spring dispatch-servlet.xml to map the JS files to your jawrJsController
      <bean id="urlMapping" class="org.springframework.web.servlet.ahdnerl.SampleUrlHandlerMapping">
         <property name="mappings">
            <props>
               <prop key="/jawr/**/*.js">jawrControllerForJs</prop>
            </props> 
         </property>
      </bean>
      
      
 4. Setup your jawr.properties
       jawr.js.bundle.XXX.id=/bundles/all.js
       jawr.js.bundle.XXX.mappings = \
           /resources/stuff/file1.js,\  
           /resources/stuff/files2.js,\
