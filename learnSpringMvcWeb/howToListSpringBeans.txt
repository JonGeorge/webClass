How to List All Spring Beans
----------------------------

In your contextInitialized() method


public class MyAppWatcher implements ServletContextListener
{
  @Override
  public void contextInitialized(ServletContextEvent aServletContextEvent)
  {
     logger.debug("contextInitialized() started");
     
     listSpringBeans(aServletContextEvent);
     
     logger.debug("contextInitialized() finished");
  }
  
  
  private listSpringBeans(ServletContextEvent aServletContextEvent)
  {
    ServletContext srvContext = aServletContextEvent.getServletContext();
    
     WebApplicationContext springAppContext = WebApplicationContextUtils.getWebApplicationContext(srvContext);
     
     String[] allSpringBeanNames = springAppContext.getBeanDefinitionNames();
     
     for (String sBeanName: allSpringBeanNames)
      {
         String sBeanClassInfo = springAppContext.getBean(sBeanName).getClass().toString();
         logger.debug("{} -> {}", sBeanName, sBeanClassInfo);
      }
  }    
  }
