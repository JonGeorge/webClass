Lesson:  How to Add a Graph using HighCharts
--------------------------------------------


Procedure
---------
 1. Install the highcharts NPM package
    unix> cd ~/intellijProjects/angularApp1/frontend
	unix> npm install highcharts
    
	You should see something like this in your frontend/package.json
		    "highcharts": "8.1.2",
			

 2. Install the highcharts wrapper NPM package
    unix> npm install highcharts-angular
	
	You should see something like this in your package.json
 		    "highcharts-angular": "2.7.0",
	
	
 3. Create the analytics/chart1 component
    unix> cd ~/intellijProjects/angularApp1/frontend
	unix> ng generate component analytics/chart1
	

 4. Setup a left and right side of the chart1.component.html
    a. Edit chart1.component.html
	b. Replace its contents with this:

		<mat-card>
		  <mat-card-title>Chart 1</mat-card-title>

		  <mat-card-content>

			<div fxFlex fxLayout="column" fxLayoutGap="0">
			  <!-- Left Side of the Page -->
			  <div fxFlex fxLayoutAlign="left center" fxLayout="row" fxLayoutGap="0">
				Left
			  </div>

			  <!-- Right Side of the Page -->
			  <div fxFlex fxLayoutAlign="end center">
				Right
			  </div>

			</div>

		  </mat-card-content>
		</mat-card>


 5. Add the HighchartsChartComponent to your app.module.ts declarations
    a. Edit your app.module.ts
	b. Add the import:
	    	import {HighchartsChartComponent} from "highcharts-angular";

 	c. Add this to your declarations: section
	   		HighchartsChartComponent,

 6. Add a charge declaration to the left side of the page
    a. Edit chart1.component.html
	b. Change the left side to this:
		  <!-- Left Side of the Page -->
		  <div fxFlex fxLayoutAlign="left center" fxLayout="row" fxLayoutGap="0">

			<highcharts-chart
			  [Highcharts] = "highcharts"
			  [options] = "chartOptions"
			  style = "width: 100%; height: 400px; display: block;">
			</highcharts-chart>
		  </div>	
		  
 7. Change chart1.component.ts so that it provides data to the chart
    a. Edit chart1.component.ts
	b. Replace it with this:

			import { Component, OnInit } from '@angular/core';
			import * as Highcharts from 'highcharts';

			@Component({
			  selector: 'app-chart1',
			  templateUrl: './chart1.component.html',
			  styleUrls: ['./chart1.component.css']
			})
			export class Chart1Component implements OnInit {

			  highcharts = Highcharts;

			  data = [{
				name: 'ItSolutionStuff.com',
				data: [500, 700, 555, 444, 777, 877, 944, 567, 666, 789, 456, 654]
			  },{
				name: 'Nicesnippets.com',
				data: [677, 455, 677, 877, 455, 778, 888, 567, 785, 488, 567, 654]
			  }];


			  chartOptions = {
				chart: {
				  type: "spline"
				},
				title: {
				  text: "Monthly Site Visitor"
				},
				xAxis:{
				  categories:["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"]
				},
				yAxis: {
				  title:{
					text:"Visitors"
				  }
				},

				series: this.data
			  };


			  constructor() { }

			  ngOnInit(): void {
			  }

			}
