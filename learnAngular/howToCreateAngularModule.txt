Procedure
---------
 1.  
 
 
 The proper way to create a module is to use the angular.module method passing at least the module name, and an array naming all the dependencies of the module  (or an empty array if there are no dependencies).
 
 This code creates 3 modules:
  
   angular.module("patientApp.Services", []);
   angular.module("patientApp.Controllers", []);   
   angular.module("patientApp", ["patientApp.Services",
                                 "patientApp.Controllers"]);
                                 
 
 
 The following code defines the three modules, 
 then comes back and adds a "run block" to the patientApp module (a run block is a piece of code that Angular will invoke once the module is assembled and configured).
 
 (function () {
    "use strict";
 
    angular.module("patientApp.Services", []);
    angular.module("patientApp.Controllers", []);   
    angular.module("patientApp", ["patientApp.Services",
                                  "patientApp.Controllers"]);
 
    var app = angular.module("patientApp");
    app.run(["$rootScope", function ($rootScope) {
        $rootScope.patientAppVersion = "0.0.0.1";
    }]);
     
}());  



Calling angular.module with just the module name will give you back a reference to an existing module 
-- So you can continue adding run blocks, services, directives, and other components into the module.                              




For now, think of the factory method as "registering a service" in the module.
Define an "alerter" service
(function () {
    var alerter = function () {
        return {
            alert: function (message) {
                // ... do something alerty
            }
        };
    };
 
    angular.module("patientApp.Services")          
           .factory("alerter", alerter);
}());


Define a "logger" service
(function() {
 
    var logger = function() {
       
        var log = function(message) {
            // .. do something loggy
        };
 
        return {
            log: log,            
        };
    };
 
    angular.module("patientApp.Services")
           .factory("logger", logger);           
 
}());


**OR** define both services
(function() {
    "use strict";
 
    var alerter = function () {
        //...
    };
 
    var logger = function() {
        //...
    };
 
    angular.module("patientApp.Services")
           .factory("alerter", alerter)
           .factory("logger", logger);           
 
}());
